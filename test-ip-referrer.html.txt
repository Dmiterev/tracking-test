  .then(data=> {

      <p>🔍 IP: <b>${data.ip}</b></p>
      <p>🌐 Referrer: <b>${document.referrer || 'прямой заход'}</b></p>
      <p>👨‍💻 User-Agent: <b>${navigator.userAgent}</b></p>
      <p>📱 Язык: <b>${navigator.language}</b></p>
    `;
  });
</script>
